优化之前程序的DAG

import org.apache.spark.{SparkConf, SparkContext}

object moveRating {
  def main(args: Array[String]): Unit = {
    val conf =new SparkConf().setAppName("movieRating").setMaster("local[*]")
    val sc=new SparkContext(conf)
    val ratings = sc.textFile(path="/home/bigdata3/ml-1m/ratings.dat")
      .map(x=>x.split("::"))
      .map(x=>(x(1),x(2)))
    val sumRatings=ratings
      .map(x => (x._1, x._2.toDouble))
      .reduceByKey(_ + _)
    val ratingsCount=ratings
      .map(x => (x._1, 1.toDouble))
      .reduceByKey(_ + _)
    val sumRatingsCounts=sumRatings.join(ratingsCount)
    val avgRatings=sumRatingsCounts.map(x=>(x._1,x._2._1/x._2._2))

    val movies = sc.textFile(path = "/home/bigdata3/ml-1m/movies.dat")
      .map(x => x.split("::"))
      .map(x => (x(0), x(1)))
      .join(avgRatings)
      .map(x=>(x._2._2,(x._1,x._2._1)))
      .sortByKey(false)
    movies.take(30).foreach(println)
  }
}

import org.apache.spark.{SparkConf, SparkContext}

object moveRating {
  def main(args: Array[String]): Unit = {
    val conf =new SparkConf().setAppName("movieRating").setMaster("local[*]")
    val sc=new SparkContext(conf)
    val avgRatings = sc.textFile(path="/home/bigdata3/ml-1m/ratings.dat")
      .map(x=>x.split("::"))
      .map(x=>(x(1),(x(2).toDouble,1.toDouble)))
      .reduceByKey((x,y)=>(x._1+y._1,x._2+y._2))
      .map(x=>(x._1,x._2._1/x._2._2))
    
    val movies = sc.textFile(path = "/home/bigdata3/ml-1m/movies.dat")
      .map(x => x.split("::"))
      .map(x => (x(0), x(1)))
      .join(avgRatings)
      .map(x=>(x._2._2,(x._1,x._2._1)))
      .sortByKey(false)
    movies.take(30).foreach(println)
    Thread.sleep(1000*60*5)
    sc.stop()
  }
}

.reduceByKey((x,y)=>(x._1+y._1,x._2+y._2))

import org.apache.spark.{SparkConf, SparkContext}

object moveRating {
  def main(args: Array[String]): Unit = {
    val conf =new SparkConf().setAppName("movieRating").setMaster("local[*]")
    val sc=new SparkContext(conf)
    val avgRatings = sc.textFile(path="/home/bigdata3/ml-1m/ratings.dat")
      .map(x=>x.split("::"))
      .map(x=>(x(1),(x(2).toDouble,1.toDouble)))
      .reduceByKey((x,y)=>(x._1+y._1,x._2+y._2))
      .filter(x=>x._2._2>20)
      .map(x=>(x._1,x._2._1/x._2._2))

    val movies = sc.textFile(path = "/home/bigdata3/ml-1m/movies.dat")
      .map(x => x.split("::"))
      .map(x => (x(0), x(1)))
      .join(avgRatings)
      .map(x=>(x._2._2,(x._1,x._2._1)))
      .sortByKey(false)
    movies.take(30).foreach(println)
    Thread.sleep(1000*60*5)
    sc.stop()
  }
}
